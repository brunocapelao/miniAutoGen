```python
"""Pipeline execution service for the CLI.

Resolves pipeline targets from project config and executes them
via the public SDK surface (miniautogen.api).
"""
from __future__ import annotations

import importlib
from typing import Any, Callable


def resolve_pipeline_target(target_str: str) -> Callable[..., Any]:
    """Resolve a 'module.path:callable_name' string to a Python callable.

    Args:
        target_str: Import path in the format 'module.path:callable_name'.

    Returns:
        The resolved callable.

    Raises:
        ValueError: If the target string format is invalid.
        ImportError: If the module cannot be imported.
        AttributeError: If the callable is not found in the module.
    """
    if ":" not in target_str or target_str.count(":") != 1:
        msg = (
            "Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    module_path, callable_name = target_str.split(":")

    if not module_path or not callable_name:
        msg = (
            "Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    try:
        module = importlib.import_module(module_path)
    except ModuleNotFoundError as exc:
        msg = f"Could not import module '{module_path}': {exc}"
        raise ImportError(msg) from exc

    if not hasattr(module, callable_name):
        msg = f"Module '{module_path}' has no attribute '{callable_name}'"
        raise AttributeError(msg)

    return getattr(module, callable_name)
```

```python
"""Pipeline execution service for the CLI.

Resolves pipeline targets from project config and executes them
via the public SDK surface (miniautogen.api).
"""
from __future__ import annotations

import importlib
import sys
from typing import Any, Callable

from miniautogen.api import (
    ExecutionPolicy,
    InMemoryEventSink,
    InMemoryRunStore,
    LoggingEventSink,
    PipelineRunner,
)


def resolve_pipeline_target(target_str: str) -> Callable[..., Any]:
    """Resolve a 'module.path:callable_name' string to a Python callable.

    Args:
        target_str: Import path in the format 'module.path:callable_name'.

    Returns:
        The resolved callable.

    Raises:
        ValueError: If the target string format is invalid.
        ImportError: If the module cannot be imported.
        AttributeError: If the callable is not found in the module.
    """
    if ":" not in target_str or target_str.count(":") != 1:
        msg = (
            "Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    module_path, callable_name = target_str.split(":")

    if not module_path or not callable_name:
        msg = (
            "Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    try:
        module = importlib.import_module(module_path)
    except ModuleNotFoundError as exc:
        msg = f"Could not import module '{module_path}': {exc}"
        raise ImportError(msg) from exc

    if not hasattr(module, callable_name):
        msg = f"Module '{module_path}' has no attribute '{callable_name}'"
        raise AttributeError(msg)

    return getattr(module, callable_name)


async def execute_pipeline(
    pipeline_target: str,
    timeout: float | None,
    verbose: bool,
    extra_sys_paths: list[str] | None = None,
) -> dict[str, Any]:
    """Execute a pipeline identified by its import target string.

    This function is the core service for the ``run`` CLI command.
    It resolves the pipeline target, sets up PipelineRunner with
    appropriate sinks, and executes the pipeline.

    Args:
        pipeline_target: Import path in 'module.path:callable_name' format.
        timeout: Optional timeout in seconds for pipeline execution.
        verbose: If True, use LoggingEventSink for detailed output.
        extra_sys_paths: Additional directories to add to sys.path for import
            resolution (e.g., the project root). Removed after execution.

    Returns:
        A dict with keys: run_id, status, output.

    Raises:
        ImportError: If the pipeline target module cannot be imported.
        AttributeError: If the pipeline target callable is not found.
        ValueError: If the pipeline target format is invalid.
        TimeoutError: If pipeline execution exceeds the timeout.
    """
    # Temporarily add extra paths to sys.path for dynamic import resolution.
    added_paths: list[str] = []
    if extra_sys_paths:
        for p in extra_sys_paths:
            if p not in sys.path:
                sys.path.insert(0, p)
                added_paths.append(p)

    try:
        builder = resolve_pipeline_target(pipeline_target)
        pipeline = builder()

        event_sink = LoggingEventSink() if verbose else InMemoryEventSink()
        run_store = InMemoryRunStore()
        execution_policy = (
            ExecutionPolicy(timeout_seconds=timeout) if timeout else None
        )

        runner = PipelineRunner(
            event_sink=event_sink,
            run_store=run_store,
            execution_policy=execution_policy,
        )

        result = await runner.run_pipeline(
            pipeline,
            {},
            timeout_seconds=timeout,
        )

        run_id = runner.last_run_id or "unknown"

        return {
            "run_id": run_id,
            "status": "finished",
            "output": result,
        }
    finally:
        for p in added_paths:
            if p in sys.path:
                sys.path.remove(p)
```

**Key design decisions:**
- `execute_pipeline` takes `pipeline_target: str` directly (not `ProjectConfig`). The command layer is responsible for extracting the target from config. This makes the service more testable and reusable.
- `extra_sys_paths` is cleaned up in a `finally` block to avoid polluting `sys.path`.
- The service imports exclusively from `miniautogen.api` (D3 compliant).

```python
"""``miniautogen run`` command -- execute a pipeline headlessly."""
from __future__ import annotations

import sys
from pathlib import Path
from typing import Any

import click

from miniautogen.cli.config import find_project_root, load_config
from miniautogen.cli.errors import PipelineNotFoundError
from miniautogen.cli.main import run_async
from miniautogen.cli.output import echo_error, echo_json, echo_success
from miniautogen.cli.services.run_pipeline import execute_pipeline


@click.command("run")
@click.argument("pipeline_name", default="main")
@click.option(
    "--timeout",
    type=float,
    default=None,
    help="Timeout in seconds for pipeline execution.",
)
@click.option(
    "--format",
    "output_format",
    type=click.Choice(["text", "json"]),
    default="text",
    help="Output format: text (default) or json.",
)
@click.option(
    "--verbose",
    is_flag=True,
    default=False,
    help="Enable verbose event logging.",
)
def run_command(
    pipeline_name: str,
    timeout: float | None,
    output_format: str,
    verbose: bool,
) -> None:
    """Execute a pipeline defined in the project config.

    PIPELINE_NAME defaults to "main" if not specified.
    """
    # Locate the project root directory.
    project_root = find_project_root(Path.cwd())
    if project_root is None:
        echo_error("No miniautogen.yaml found in current or parent directories.")
        sys.exit(2)

    # Load and validate the project config.
    config_path = project_root / "miniautogen.yaml"
    try:
        config = load_config(config_path)
    except Exception as exc:
        echo_error(f"Failed to load config: {exc}")
        sys.exit(3)

    # Look up the pipeline target in config.
    pipelines = config.pipelines or {}
    if pipeline_name not in pipelines:
        available = list(pipelines.keys())
        echo_error(
            f"No pipeline named '{pipeline_name}' in project config. "
            f"Available: {available}"
        )
        sys.exit(1)

    pipeline_target = pipelines[pipeline_name]["target"]
    extra_paths = [str(project_root)]

    # Execute the pipeline via the service layer.
    try:
        result: dict[str, Any] = run_async(
            execute_pipeline,
            pipeline_target=pipeline_target,
            timeout=timeout,
            verbose=verbose,
            extra_sys_paths=extra_paths,
        )
    except (ImportError, AttributeError, ValueError) as exc:
        echo_error(f"Failed to resolve pipeline target: {exc}")
        sys.exit(1)
    except TimeoutError:
        echo_error("Pipeline execution timed out.")
        sys.exit(1)
    except Exception as exc:
        echo_error(f"Pipeline execution failed: {exc}")
        sys.exit(1)

    # Render output.
    if output_format == "json":
        echo_json(result)
    else:
        echo_success(f"Pipeline '{pipeline_name}' completed successfully.")
        click.echo(f"  Run ID: {result['run_id']}")
        click.echo(f"  Status: {result['status']}")
```
