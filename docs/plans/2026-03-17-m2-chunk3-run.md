```python
"""Pipeline execution service for the CLI.

Resolves pipeline targets from project config and executes them
via the public SDK surface (miniautogen.api).
"""
from __future__ import annotations

import importlib
from typing import Any, Callable


def resolve_pipeline_target(target_str: str) -> Callable[..., Any]:
    """Resolve a 'module.path:callable_name' string to a Python callable.

    Args:
        target_str: Import path in the format 'module.path:callable_name'.

    Returns:
        The resolved callable.

    Raises:
        ValueError: If the target string format is invalid.
        ImportError: If the module cannot be imported.
        AttributeError: If the callable is not found in the module.
    """
    if ":" not in target_str or target_str.count(":") != 1:
        msg = (
            f"Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    module_path, callable_name = target_str.split(":")

    if not module_path or not callable_name:
        msg = (
            f"Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    try:
        module = importlib.import_module(module_path)
    except ModuleNotFoundError as exc:
        msg = f"Could not import module '{module_path}': {exc}"
        raise ImportError(msg) from exc

    if not hasattr(module, callable_name):
        msg = f"Module '{module_path}' has no attribute '{callable_name}'"
        raise AttributeError(msg)

    return getattr(module, callable_name)
```

```python
"""Pipeline execution service for the CLI.

Resolves pipeline targets from project config and executes them
via the public SDK surface (miniautogen.api).
"""
from __future__ import annotations

import importlib
import sys
from typing import TYPE_CHECKING, Any, Callable

from miniautogen.api import (
    ExecutionPolicy,
    InMemoryEventSink,
    InMemoryRunStore,
    LoggingEventSink,
    PipelineRunner,
)

if TYPE_CHECKING:
    from miniautogen.cli.config import ProjectConfig


def resolve_pipeline_target(target_str: str) -> Callable[..., Any]:
    """Resolve a 'module.path:callable_name' string to a Python callable.

    Args:
        target_str: Import path in the format 'module.path:callable_name'.

    Returns:
        The resolved callable.

    Raises:
        ValueError: If the target string format is invalid.
        ImportError: If the module cannot be imported.
        AttributeError: If the callable is not found in the module.
    """
    if ":" not in target_str or target_str.count(":") != 1:
        msg = (
            f"Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    module_path, callable_name = target_str.split(":")

    if not module_path or not callable_name:
        msg = (
            f"Pipeline target must be in 'module.path:callable_name' format, "
            f"got '{target_str}'"
        )
        raise ValueError(msg)

    try:
        module = importlib.import_module(module_path)
    except ModuleNotFoundError as exc:
        msg = f"Could not import module '{module_path}': {exc}"
        raise ImportError(msg) from exc

    if not hasattr(module, callable_name):
        msg = f"Module '{module_path}' has no attribute '{callable_name}'"
        raise AttributeError(msg)

    return getattr(module, callable_name)


async def execute_pipeline(
    config: ProjectConfig,
    pipeline_name: str,
    timeout: float | None,
    verbose: bool,
    extra_sys_paths: list[str] | None = None,
) -> dict[str, Any]:
    """Execute a pipeline defined in the project config.

    Args:
        config: The loaded project configuration.
        pipeline_name: Name of the pipeline to execute (key in config.pipelines).
        timeout: Optional timeout in seconds for pipeline execution.
        verbose: If True, use LoggingEventSink for detailed output.
        extra_sys_paths: Additional directories to add to sys.path for import
            resolution (e.g., the project root). Removed after execution.

    Returns:
        A dict with keys: run_id, status, output.

    Raises:
        KeyError: If pipeline_name is not found in config.pipelines.
        ImportError: If the pipeline target module cannot be imported.
        AttributeError: If the pipeline target callable is not found.
    """
    if pipeline_name not in config.pipelines:
        msg = (
            f"No pipeline named '{pipeline_name}' in project config. "
            f"Available: {list(config.pipelines.keys())}"
        )
        raise KeyError(msg)

    pipeline_config = config.pipelines[pipeline_name]
    target_str = pipeline_config.target

    # Temporarily add extra paths to sys.path for dynamic import resolution.
    added_paths: list[str] = []
    if extra_sys_paths:
        for p in extra_sys_paths:
            if p not in sys.path:
                sys.path.insert(0, p)
                added_paths.append(p)

    try:
        builder = resolve_pipeline_target(target_str)
        pipeline = builder()

        event_sink = LoggingEventSink() if verbose else InMemoryEventSink()
        run_store = InMemoryRunStore()
        execution_policy = (
            ExecutionPolicy(timeout_seconds=timeout) if timeout else None
        )

        runner = PipelineRunner(
            event_sink=event_sink,
            run_store=run_store,
            execution_policy=execution_policy,
        )

        result = await runner.run_pipeline(
            pipeline,
            {},
            timeout_seconds=timeout,
        )

        run_id = runner.last_run_id or "unknown"

        return {
            "run_id": run_id,
            "status": "finished",
            "output": result,
        }
    finally:
        for p in added_paths:
            if p in sys.path:
                sys.path.remove(p)
```

```python
"""CLI command: miniautogen run <pipeline>.

Executes a pipeline defined in the project config headlessly.
"""
from __future__ import annotations

import json
import os
import sys
from typing import Any

import click

from miniautogen.cli.config import load_project_config
from miniautogen.cli.errors import CLIError
from miniautogen.cli.main import run_async
from miniautogen.cli.output import render_result
from miniautogen.cli.services.run_pipeline import execute_pipeline


@click.command("run")
@click.argument("pipeline_name", default="main")
@click.option(
    "--timeout",
    type=float,
    default=None,
    help="Timeout in seconds for pipeline execution.",
)
@click.option(
    "--format",
    "output_format",
    type=click.Choice(["text", "json"]),
    default="text",
    help="Output format: text (default) or json.",
)
@click.option(
    "--verbose",
    is_flag=True,
    default=False,
    help="Enable verbose event logging.",
)
def run_command(
    pipeline_name: str,
    timeout: float | None,
    output_format: str,
    verbose: bool,
) -> None:
    """Execute a pipeline defined in the project config.

    PIPELINE_NAME defaults to "main" if not specified.
    """
    try:
        config = load_project_config()
    except CLIError as exc:
        click.echo(f"Error: {exc}", err=True)
        raise SystemExit(1) from exc

    # Add the project root to sys.path so pipeline modules can be imported.
    project_dir = os.environ.get("MINIAUTOGEN_PROJECT_DIR", os.getcwd())
    extra_paths = [project_dir]

    try:
        result = run_async(
            execute_pipeline(
                config=config,
                pipeline_name=pipeline_name,
                timeout=timeout,
                verbose=verbose,
                extra_sys_paths=extra_paths,
            )
        )
    except KeyError as exc:
        click.echo(f"Error: {exc}", err=True)
        raise SystemExit(1) from exc
    except (ImportError, AttributeError, ValueError) as exc:
        click.echo(f"Error resolving pipeline target: {exc}", err=True)
        raise SystemExit(1) from exc
    except TimeoutError:
        click.echo("Error: Pipeline execution timed out.", err=True)
        raise SystemExit(1)
    except Exception as exc:
        click.echo(f"Error during pipeline execution: {exc}", err=True)
        raise SystemExit(1) from exc

    render_result(result, output_format)
```

```python
def render_result(data: dict, fmt: str = "text") -> None:
    """Render a result dict to stdout in the given format."""
    if fmt == "json":
        click.echo(json.dumps(data, indent=2, default=str))
    else:
        for key, value in data.items():
            click.echo(f"{key}: {value}")
```
